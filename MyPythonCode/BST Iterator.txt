# Definition for a  binary tree node
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class BSTIterator(object):
    def __init__(self, root):
        """
        :type root: TreeNode
        """
        self.stack=[]
        self.push(root)

    def hasNext(self):
        """
        :rtype: bool
        """
        return self.stack
        

    def next(self):
        """
        :rtype: int
        """
        tmp=self.stack.pop()
        self.push(tmp.right)
        return tmp.val
        
        
    
    def push(self,node):
        while node!=None:
            self.stack.append(node)
            node=node.left
